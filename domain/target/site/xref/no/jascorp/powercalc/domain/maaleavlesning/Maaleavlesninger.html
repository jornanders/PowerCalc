27  	public void addMaaleavlesning(Maaleavlesning maaleavlesning) {
28  		if (maaleavlesning == null) {
29  			throw new IllegalArgumentException("Måleavlesning kan ikke være null");
30  		}
31  		
32  		LinkedList<Maaleavlesning> avlesninger = getAvlesningerForMaalepunkt(maaleavlesning.getPunkt());
33  		avlesninger.add(maaleavlesning);
34  		Collections.sort(avlesninger);
35  		assureStandIsIncreasing(avlesninger);
36  	}

38  	private LinkedList<Maaleavlesning> getAvlesningerForMaalepunkt(Maalepunkt maalepunkt) {
39  		if (!maaleavlesninger.containsKey(maalepunkt)) {
40  			maaleavlesninger.put(maalepunkt, new LinkedList<Maaleavlesning>());
41  		}
42  		return maaleavlesninger.get(maalepunkt);
43  	}

45  	private void assureStandIsIncreasing(LinkedList<Maaleavlesning> avlesninger) {
46  		Maaleavlesning forrige = null;
47  		for (Maaleavlesning avlesning : avlesninger) {
48  			if (forrige != null) {
49  				if (forrige.getStand() > avlesning.getStand()) {
50  					throw new IllegalArgumentException("Stand som er lagt til er lavere enn forrige stand");
51  				}
52  			}
53  			forrige = avlesning;
54  		}
55  	}

57  	public double beregnForbrukForIntervall(final Datointervall datointervall, final Maalepunkt maalepunkt) {
58  		MaaleavlesningIntervall intervaller = getIntervaller(datointervall, maalepunkt);
59  		double forbruk = 0;
60  		while (intervaller != null && intervaller != MaaleavlesningIntervall.NULL_INTERVALL) {
61  			forbruk += intervaller.getForbruk(datointervall);
62  			intervaller = intervaller.getNesteIntervall();
63  		}
64  		Logger.getLogger(getClass()).debug("Totalt forbruk for, " + maalepunkt + ", intervall (" + datointervall + ") : " + forbruk);
65  		return forbruk;
66  	}

68  	public MaaleavlesningIntervall getIntervaller(Maalepunkt maalepunkt) {
69  		Maaleavlesning forrige = null;
70  		MaaleavlesningIntervall forrigeIntervall = null;
71  		for (Maaleavlesning avlesning : getAvlesningerForMaalepunkt(maalepunkt)) {
72  			if (forrige != null) {
73  				forrigeIntervall = new MaaleavlesningIntervall(forrige, avlesning, forrigeIntervall);
74  			}
75  			forrige = avlesning;
76  		}
77  		return forrigeIntervall != null ? forrigeIntervall.getForsteIntervall() : MaaleavlesningIntervall.NULL_INTERVALL;
78  	}

80  	public MaaleavlesningIntervall getIntervaller(Datointervall datointervall, Maalepunkt maalepunkt) {
81  		MaaleavlesningIntervall intervaller = getIntervaller(maalepunkt);
82  		//sjekker om datointervallet ligger før hele kjeden med avlesningsintervall => bruker det første intervallet for hele beregningen
83  		if (isTilDatoForDatointervallBeforeForsteAvlesning(datointervall, intervaller)) {
84  			intervaller.setNesteIntervall(null);
85  			return intervaller;
86  		}
87  		//sjekker om datointervallet ligger etter hele kjeden med avlesningsintervall => bruker det siste intervallet for hele beregningen
88  		if (isFraDatoForDatointervallAfterSisteAvlesning(datointervall, intervaller)) {
89  			intervaller = intervaller.getSisteIntervall();
90  			intervaller.setForrigeIntervall(null);
91  			return intervaller;
92  		}
93  		
94  		MaaleavlesningIntervall subIntervaller = intervaller;
95  		//finner start-intervallet
96  		while (subIntervaller != null 
97  				&& !datointervall.getFraDato().before(subIntervaller.getAvlesningDatointervall().getFraDato()) //at fra dato ligger får første måling
98  				&& !subIntervaller.getAvlesningDatointervall().isDateInRange(datointervall.getFraDato())) {
99  			subIntervaller = subIntervaller.getNesteIntervall();
100 			subIntervaller.setForrigeIntervall(null);
101 		}
102 		
103 		if (subIntervaller == null) {
104 			subIntervaller = getIntervaller(maalepunkt);
105 		}
106 		
107 		MaaleavlesningIntervall sisteIntervall = subIntervaller.getSisteIntervall();
108 		//finner slutt-intervallet
109 		while (sisteIntervall.getForrigeIntervall() != null
110 				&& !datointervall.getTilDato().after(sisteIntervall.getAvlesningDatointervall().getTilDato()) //at til dato ligger etter siste måling
111 				&& !sisteIntervall.getAvlesningDatointervall().isDateInRange(datointervall.getTilDato())) {
112 			sisteIntervall = sisteIntervall.getForrigeIntervall();
113 		}
114 		
115 		sisteIntervall.setNesteIntervall(null);
116 		return sisteIntervall.getForsteIntervall();
117 	}
